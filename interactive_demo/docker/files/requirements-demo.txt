# Minimal extra deps for the interactive demo on top of the public PyTorch CUDA image.
#
# IMPORTANT — keep NumPy < 2.x.
# pytorch/pytorch:2.5.1-cuda12.1-cudnn9-runtime ships NumPy 2.1.2, but most of the
# wheels we depend on (opencv-python 4.8.x, scipy/matplotlib at the versions pinned
# below, natten 0.17.5+torch250cu121) were compiled against NumPy 1.x. Mixing them
# yields:  AttributeError: _ARRAY_API not found / numpy.core.multiarray failed to import
# when `import cv2` runs. Forcing NumPy 1.26.x is the conservative fix.
numpy<2
setuptools<81

# Gradio 4.44.x calls Starlette's TemplateResponse with the OLD positional signature
# (name, context). Starlette >= 0.45 changed it to (request, name, ...), which breaks
# Gradio with: TypeError: unhashable type: 'dict' (jinja2 cache_key is a dict).
# Pin compatible web stack:
gradio==4.44.1
gradio_client==1.3.0
fastapi==0.112.2
starlette==0.38.6
jinja2==3.1.4
uvicorn==0.30.6
httpx==0.27.2
pydantic==2.9.2
